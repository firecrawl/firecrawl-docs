```bash cURL
# 无需 API 密钥即可开始使用——添加 -H "Authorization: Bearer fc-YOUR_API_KEY" 可获得更高限流：
curl -X POST https://api.firecrawl.dev/v2/search \
  -H "Content-Type: application/json" \
  -d '{
    "query": "firecrawl web scraping",
    "limit": 3,
    "scrapeOptions": {
      "formats": ["markdown", "links"]
    }
  }'
```
