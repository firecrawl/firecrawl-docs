```js Node
import Firecrawl from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: "fc-你的 API 密钥" });

const schema = {
  type: 'object',
  properties: {
    title: { type: 'string' }
  },
  required: ['title']
};

const res = await firecrawl.extract({
  urls: ['https://docs.firecrawl.dev'],
  prompt: '提取页面标题'
  schema,
  scrapeOptions: { formats: [{ type: 'json', prompt: '提取', schema }] }
});

console.log(res.status || res.success, res.data);
```
