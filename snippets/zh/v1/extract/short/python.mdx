from pydantic import BaseModel

class ExtractSchema(BaseModel):
    company_mission: str

# 定义要提取数据的 URL
urls = [
    'https://firecrawl.dev/*'
]

# 定义用于提取的提示
prompt = '从页面提取公司使命。'

# 执行提取操作
scrape_result = app.extract(urls, prompt=prompt, schema=ExtractSchema.model_json_schema())

# 检查提取是否成功
if not scrape_result.success:
    raise Exception(f"提取失败：{scrape_result.error}")

# 输出提取结果
print(scrape_result.data)
