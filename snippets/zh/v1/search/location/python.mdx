```python Python
from firecrawl import FirecrawlApp

# 使用你的 API 密钥初始化客户端
app = FirecrawlApp(
  # 开始使用时无需 API 密钥；如需更高的速率限制，请添加一个：
  # api_key="fc-YOUR_API_KEY",
)

# 使用位置设置进行搜索（德国）
search_result = app.search(
    "web scraping tools",
    limit=5,
    location="Germany"
)

# 处理结果
for result in search_result.data:
    print(f"标题: {result['title']}")
    print(f"URL: {result['url']}") 
```
