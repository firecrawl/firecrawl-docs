```js Node
import FirecrawlApp from 'firecrawl';

// 初始化客户端
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// 定义研究参数
const params = {
  maxDepth: 5,  // 研究迭代次数
  timeLimit: 180,  // 时间上限（秒）
  maxUrls: 15  // 待分析的 URL 最大数量
};

// 启动研究并接收实时更新
const onActivity = (activity) => {
  console.log(`[${activity.type}] ${activity.message}`);
};

// 运行深度研究
const results = await firecrawl.deepResearch(
  '量子计算领域的最新进展是什么？',
  params,
  onActivity
);

// 查看研究结果
console.log(`最终分析：${results.data.finalAnalysis}`);
console.log(`来源：${results.data.sources.length} 个参考`); 
```
