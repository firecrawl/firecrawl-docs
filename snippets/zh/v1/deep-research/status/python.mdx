```python Python
from firecrawl import FirecrawlApp

# 初始化客户端
firecrawl = FirecrawlApp(api_key="your_api_key")

# 检查研究任务状态
status = firecrawl.check_deep_research_status("job_id")

# 输出当前进度
print(f"状态: {status.status}")
print(f"进度: {status.current_depth}/{status.max_depth} 次迭代")

if status.status == 'completed':
    print(f"最终分析: {status.data.final_analysis}")
    print(f"找到的来源数: {len(status.data.sources)}") 
```
