```rust Rust
use firecrawl::FirecrawlApp;
use std::collections::HashMap;

#[tokio::main]
async fn main() {
    // 使用你的 API 密钥初始化 FirecrawlApp
    let app = FirecrawlApp::new("fc-YOUR_API_KEY").expect("初始化 FirecrawlApp 失败");

    // 如果缓存的数据不超过 1 小时（3600000 ms），则使用缓存
    // 这比全量新抓取快 500%！
    let mut scrape_params = HashMap::new();
    scrape_params.insert("formats", vec!["markdown"]);
    scrape_params.insert("maxAge", 3600000); // 1 小时（毫秒）

    let scrape_result = app
        .scrape_url("https://firecrawl.dev", Some(scrape_params))
        .await
        .expect("抓取该 URL 失败");

    println!("{}", scrape_result.markdown.unwrap_or_default());
}
```
