```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
  # 无需 API 密钥即可开始使用——添加一个以获得更高限流：
  # api_key="fc-YOUR_API_KEY",
)

# 若缓存数据的时间小于 1 小时（3600000 毫秒），则使用缓存
# This can be lower latency when a recent cache entry is available.
scrape_result = app.scrape_url(
    'https://firecrawl.dev', 
    params={
        'formats': ['markdown'],
        'max_age': 3600000  # 1 小时（毫秒）
    }
)

print(scrape_result['markdown'])
```
