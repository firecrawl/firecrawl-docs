```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# 若缓存数据的时间小于 1 小时（3600000 毫秒），则使用缓存
# 这比全量重新抓取可快达 500%！
scrape_result = app.scrape_url(
    'https://firecrawl.dev', 
    params={
        'formats': ['markdown'],
        'maxAge': 3600000  # 1 小时（毫秒）
    }
)

print(scrape_result['markdown'])
```
