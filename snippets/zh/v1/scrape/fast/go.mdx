```go Go 语言
package main

import (
    "fmt"
    "log"

    "github.com/mendableai/firecrawl-go"
)

func main() {
    app, err := firecrawl.NewFirecrawlApp("fc-YOUR_API_KEY")
    if err != nil {
        log.Fatalf("FirecrawlApp 初始化失败：%v", err)
    }

    // 若缓存数据在 1 小时内（3600000 毫秒），则直接使用缓存
    // 这比全量重新抓取快 500%！
    scrapeParams := map[string]interface{}{
        "formats": []string{"markdown"},
        "maxAge":  3600000, // 1 小时（毫秒）
    }

    scrapeResult, err := app.ScrapeURL("https://firecrawl.dev", scrapeParams)
    if err != nil {
        log.Fatalf("抓取 URL 失败：%v", err)
    }

    fmt.Println(scrapeResult.Markdown)
}
```
