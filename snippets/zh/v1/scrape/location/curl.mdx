```bash cURL
# 无需 API 密钥即可开始使用 — 添加 -H "Authorization: Bearer YOUR_API_KEY" 可获得更高限流上限：
curl -X POST https://api.firecrawl.dev/v1/scrape \
    -H 'Content-Type: application/json' \
    -d '{
        "url": "airbnb.com",
        "formats": ["markdown"],
        "location": {
            "country": "BR",
            "languages": ["pt-BR"]
        }
    }'
```
