```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
  # 无需 API 密钥即可开始使用 — 添加一个以获得更高的限流额度：
  # api_key="fc-YOUR_API_KEY",
)

# Scrape a website:
scrape_result = app.scrape_url('airbnb.com', 
    formats=['markdown', 'html'], 
    location={
        'country': 'BR',
        'languages': ['pt-BR']
    }
)
print(scrape_result)
```
