```js Node（Node.js）
// 安装：npm install @mendable/firecrawl-js

import { FirecrawlApp } from '@mendable/firecrawl-js';

const app = new FirecrawlApp({ apiKey: 'YOUR_API_KEY' });

// 带重试逻辑的抓取函数
async function scrapeWithRetry(url) {
  try {
    // 先用默认代理尝试
    const content = await app.scrapeUrl(url);
    
    // 检查是否为错误状态码
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`收到状态码 ${statusCode}，改用隐匿代理重试`);
      // 使用隐匿代理重试
      return await app.scrapeUrl(url, {
        proxy: 'stealth'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`错误：${error.message}`);
    // 捕获异常后使用隐匿代理重试
    try {
      return await app.scrapeUrl(url, {
        proxy: 'stealth'
      });
    } catch (retryError) {
      console.error(`隐匿代理也失败：${retryError.message}`);
      throw retryError;
    }
  }
}

// 使用方式
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
