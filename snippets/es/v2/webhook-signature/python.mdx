```python Python/Flask
import hmac
import hashlib
from flask import Flask, request, abort

app = Flask(__name__)

WEBHOOK_SECRET = 'your-webhook-secret-here'  # Consíguelo en el panel de Firecrawl

@app.post('/webhook/firecrawl')
def webhook():
    signature = request.headers.get('X-Firecrawl-Signature')
    
    if not signature:
        abort(401, 'Falta el encabezado de la firma')
    
    # Extrae el hash del encabezado de la firma
    try:
        algorithm, hash_value = signature.split('=', 1)
        if algorithm != 'sha256':
            abort(401, 'Algoritmo de firma inválido')
    except ValueError:
        abort(401, 'Formato de firma inválido')
    
    # Calcula la firma esperada
    expected_signature = hmac.new(
        WEBHOOK_SECRET.encode('utf-8'),
        request.data,
        hashlib.sha256
    ).hexdigest()
    
    # Verifica la firma usando una comparación segura frente a ataques de tiempo
    if not hmac.compare_digest(hash_value, expected_signature):
        abort(401, 'Firma inválida')
    
    # Analiza y procesa el webhook verificado
    event = request.get_json(force=True)
    print('Webhook de Firecrawl verificado:', event)
    
    return 'ok', 200

if __name__ == '__main__':
    app.run(port=3000)
```
