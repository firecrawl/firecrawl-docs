```javascript JavaScript
import Firecrawl from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: "fc-YOUR_API_KEY" });

// Rastreo con scraping en caché: hasta 500% más rápido en páginas vistas recientemente
const crawlResult = await firecrawl.crawl('https://firecrawl.dev', {
  limit: 100,
  scrapeOptions: {
    formats: ['markdown'],
    maxAge: 3600000 // Usar datos en caché si tienen menos de 1 hora de antigüedad
  }
});

crawlResult.data.forEach(page => {
  console.log(`URL: ${page.metadata.sourceURL}`);
  console.log(`Contenido: ${page.markdown.substring(0, 200)}...`);
});
```
