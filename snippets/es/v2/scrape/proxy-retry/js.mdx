```js Node
// npm install @mendable/firecrawl-js

import { Firecrawl } from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: 'YOUR_API_KEY' });

// Función para hacer scraping con lógica de reintentos
async function scrapeWithRetry(url) {
  try {
    // Primer intento con el proxy predeterminado
    const content = await firecrawl.scrape(url);
    
    // Comprobar si recibimos un código de estado de error
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`Se obtuvo el código de estado ${statusCode}; reintentando con el proxy enhanced`);
      // Reintentar con el proxy enhanced
      return await firecrawl.scrape(url, {
        proxy: 'enhanced'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`Error: ${error.message}`);
    // Reintentar con el proxy enhanced ante una excepción
    try {
      return await firecrawl.scrape(url, {
        proxy: 'enhanced'
      });
    } catch (retryError) {
      console.error(`El proxy enhanced también falló: ${retryError.message}`);
      throw retryError;
    }
  }
}

// Uso
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
