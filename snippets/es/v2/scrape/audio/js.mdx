```js Node
import { Firecrawl } from 'firecrawl';

const firecrawl = new Firecrawl({
  // No se necesita API key para comenzar — añade una para límites de tasa más altos:
  // apiKey: "fc-YOUR-API-KEY",
});

const doc = await firecrawl.scrape('https://www.youtube.com/watch?v=dQw4w9WgXcQ', {
  formats: ['audio']
});

console.log(doc.audio); // URL firmada de GCS para el archivo MP3
```
