```python Python
from firecrawl import Firecrawl

firecrawl = Firecrawl(
  # No se necesita API key para empezar — agrega una para obtener límites de tasa más altos:
  # api_key='fc-YOUR_API_KEY',
)

result = firecrawl.scrape(
    url='https://example.com/restaurants/the-bistro/menu',
    formats=['markdown', 'menu']
)

print(result['markdown'])
print(result['menu'])
```
