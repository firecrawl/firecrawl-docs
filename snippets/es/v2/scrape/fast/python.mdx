```python Python
from firecrawl import Firecrawl

firecrawl = Firecrawl(api_key="fc-YOUR_API_KEY")

# Usa datos en caché si tienen menos de 1 hora (3600000 ms)
# ¡Esto puede ser hasta un 500% más rápido que hacer un scrape desde cero!
scrape_result = firecrawl.scrape(
    'https://firecrawl.dev',
    formats=['markdown'],
    max_age=3600000  # 1 hour in milliseconds
)

print(scrape_result.markdown)
```
