```js Node
import { Firecrawl } from 'firecrawl';

const firecrawl = new Firecrawl({
  // No se necesita API key para comenzar — agrega una para límites de tasa más altos:
  // apiKey: "fc-TU-API-KEY",
});

const doc = await firecrawl.scrape('https://example.com', {
  formats: [{ type: 'screenshot', fullPage: true, quality: 80, viewport: { width: 1280, height: 800 } }]
});

console.log(doc.screenshot);
```
