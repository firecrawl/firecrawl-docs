```python Python
from firecrawl import FirecrawlApp

# Inicializar el cliente
firecrawl = FirecrawlApp(api_key="your_api_key")

# Crear tarea asíncrona
job = firecrawl.async_generate_llms_text(
    url="https://example.com",
)

if job.success:
    job_id = job.id

# Comprobar el estado de la generación de LLMs.txt
status = firecrawl.check_generate_llms_text_status("job_id")

# Imprimir el estado actual
print(f"Estado: {status.status}")

if status.status == 'completed':
    print("Contenido de LLMs.txt:", status.data.llmstxt)
    if 'llmsfulltxt' in status.data:
        print("Contenido del texto completo:", status.data.llmsfulltxt)
    print(f"URLs procesadas: {len(status.data.processed_urls)}")
```
