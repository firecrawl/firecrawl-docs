```js Node.js
import FirecrawlApp from "firecrawl";

// Inicializar el cliente
const firecrawl = new FirecrawlApp({ apiKey: "your_api_key" });

// Crear una tarea asíncrona
const job = await firecrawl.asyncGenerateLLMsText("https://example.com", {
  maxUrls: 50,
  showFullText: true
});

// Comprobar el estado de la generación de LLMs.txt
const status = await firecrawl.checkGenerateLLMsTextStatus(job.jobId);

// Imprimir el estado actual
console.log(`Estado: ${status.status}`);

if (status.status === "completed") {
  console.log("Contenido de LLMs.txt:", status.data.llmstxt);
  if (status.data.llmsfulltxt) {
    console.log("Contenido del texto completo:", status.data.llmsfulltxt);
  }
  console.log(`URL procesadas: ${status.data.processedUrls.length}`);
}
```
