```python Python
from firecrawl import FirecrawlApp

# Inicializa el cliente
firecrawl = FirecrawlApp(api_key="your_api_key")

# Genera llms.txt con sondeo
results = firecrawl.generate_llms_text(
    url="https://example.com",
    max_urls=2,
    show_full_text=True
)

# Accede a los resultados de la generaci√≥n
if results.success:
    print(f"Estado: {results.status}")
    print(f"Datos generados: {results.data}")
else:
    print(f"Error: {results.error}")
```
