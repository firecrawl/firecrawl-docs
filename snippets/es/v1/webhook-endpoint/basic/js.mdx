const express = require('express');
const app = express();

app.post('/webhook', express.json(), (req, res) => {
  const { success, type, id, data, metadata, error } = req.body;
  
  switch (type) {
    case 'crawl.started':
    case 'batch_scrape.started':
      console.log(`${type.split('.')[0]} ${id} iniciado`);
      break;
      
    case 'crawl.page':
    case 'batch_scrape.page':
      if (success && data.length > 0) {
        console.log(`Página rastreada: ${data[0].metadata.sourceURL}`);
        // Procesa los datos de la página rastreada
        processScrapedPage(data[0]);
      }
      break;
      
    case 'crawl.completed':
    case 'batch_scrape.completed':
      console.log(`${type.split('.')[0]} ${id} completado correctamente`);
      break;
      
    case 'crawl.failed':
    case 'batch_scrape.failed':
      console.error(`${type.split('.')[0]} ${id} con error: ${error}`);
      break;
  }
  
  // Responde siempre con 200 para confirmar la recepción
  res.status(200).send('OK');
});
