from pydantic import BaseModel

class ExtractSchema(BaseModel):
    company_mission: str


# Define el prompt para la extracción
prompt = 'Extrae la misión de la empresa del sitio web de Firecrawl.'

# Ejecuta la extracción
scrape_result = app.extract(prompt=prompt, schema=ExtractSchema.model_json_schema())

# Comprueba si la extracción fue exitosa
if not scrape_result.success:
    raise Exception(f"Fallo al extraer: {scrape_result.error}")

# Muestra los datos extraídos
print(scrape_result.data)
