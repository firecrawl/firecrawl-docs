```js Node
import FirecrawlApp from "@mendable/firecrawl-js";

const app = new FirecrawlApp({
apiKey: "fc-YOUR_API_KEY"
});

const scrapeResult = await app.extract([
'https://nextbase.com/dash-cams/622gw-dash-cam'
], {
prompt: "Extrae detalles sobre las mejores cámaras dash, incluidos precios, características, pros y contras, y reseñas.",
enableWebSearch: true // Habilita la búsqueda web para obtener un mejor contexto
});

if (!scrapeResult.success) {
throw new Error(`Error al extraer: ${scrapeResult.error}`)
}

console.log(scrapeResult.data);
```
