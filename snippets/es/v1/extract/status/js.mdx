```js Node
import FirecrawlApp from "@mendable/firecrawl-js";

const app = new FirecrawlApp({
  apiKey: "fc-YOUR_API_KEY"
});

// Primero inicia un trabajo de extracción
const extractJob = await app.asyncExtract([
  'https://docs.firecrawl.dev/*', 
  'https://firecrawl.dev/'
], {
  prompt: "Extrae la misión de la empresa y las características de estas páginas."
});

// Obtén el estado del trabajo de extracción
const jobStatus = await app.getExtractStatus(extractJob.jobId);

console.log(jobStatus);
// Ejemplo de salida:
// {
//   status: "completed",
//   progress: 100,
//   results: [{
//     url: "https://docs.firecrawl.dev",
//     data: { ... }
//   }]
// }
```
