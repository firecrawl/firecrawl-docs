```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
    api_key="fc-YOUR_API_KEY"
)

# Primero inicia un trabajo de extracción
extract_job = app.async_extract([
    'https://docs.firecrawl.dev/*', 
    'https://firecrawl.dev/'
], prompt="Extrae la misión de la empresa y las funcionalidades de estas páginas.")

# Obtén el estado del trabajo de extracción
job_status = app.get_extract_status(extract_job.id)

print(job_status)
# Ejemplo de salida:
# id=None
# status='completed'
# expiresAt=datetime.datetime(...)
# success=True
# data=[{ ... }]
# error=None
# warning=None
# sources=None
```
