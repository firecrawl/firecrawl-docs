```js Node.js
import FirecrawlApp from '@mendable/firecrawl-js';

// Inicializa el cliente con tu clave de API
const app = new FirecrawlApp({apiKey: "fc-YOUR_API_KEY"});

// Busca y raspa contenido
app.search("firecrawl web scraping", {
  limit: 3,
  scrapeOptions: {
    formats: ["markdown", "links"]
  }
})
.then(searchResult => {
  // Procesa los resultados
  searchResult.data.forEach(result => {
    console.log(`TÃ­tulo: ${result.title}`);
    console.log(`URL: ${result.url}`);
    console.log(`Contenido: ${result.markdown?.substring(0, 150)}...`);
    console.log(`Enlaces: ${(result.links || []).slice(0, 3).join(', ')}...`);
  });
});
```
