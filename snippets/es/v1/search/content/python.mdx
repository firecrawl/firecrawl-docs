```python Python
from firecrawl import FirecrawlApp, ScrapeOptions

# Inicializa el cliente con tu clave de API
app = FirecrawlApp(
  # No se necesita clave de API para comenzar — añade una para límites de tasa más altos:
  # api_key="fc-YOUR_API_KEY",
)

# Search and scrape content
search_result = app.search(
    "firecrawl web scraping",
    limit=3,
    scrape_options=ScrapeOptions(formats=["markdown", "links"])
)

# Process the results
for result in search_result.data:
    print(f"Title: {result['title']}")
    print(f"URL: {result['url']}")
    print(f"Content: {result['markdown'][:150]}...")  # first 150 chars
    print(f"Links: {', '.join(result['links'][:3])}...")  # first 3 links
```
