```python Python
from firecrawl import FirecrawlApp, ScrapeOptions

# Inicializa el cliente con tu clave de API
app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# Busca y hace scraping de contenido
search_result = app.search(
    "firecrawl web scraping",
    limit=3,
    scrape_options=ScrapeOptions(formats=["markdown", "links"])
)

# Procesa los resultados
for result in search_result.data:
    print(f"TÃ­tulo: {result['title']}")
    print(f"URL: {result['url']}")
    print(f"Contenido: {result['markdown'][:150]}...")  # primeros 150 caracteres
    print(f"Enlaces: {', '.join(result['links'][:3])}...")  # primeros 3 enlaces
```
