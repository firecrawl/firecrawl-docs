```js Node
import { Firecrawl, SearchResponse } from 'firecrawl';

// Inicializa el cliente con tu clave de API
const app = new Firecrawl({
  // No se necesita clave de API para comenzar — añade una para obtener límites de tasa más altos:
  // apiKey: "fc-YOUR_API_KEY",
});

// Realiza una búsqueda básica
app.search("firecrawl web scraping", { limit: 5 })
  .then(searchResult => {
    // Procesa los resultados de la búsqueda
    searchResult.data.forEach(result => {
      console.log(`Title: ${result.title}`);
      console.log(`URL: ${result.url}`);
      console.log(`Description: ${result.description}`);
    });
  }); 
```
