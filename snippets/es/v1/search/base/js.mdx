```js Node
import FirecrawlApp, { SearchResponse } from '@mendable/firecrawl-js';

// Inicializa el cliente con tu clave de API
const app = new FirecrawlApp({apiKey: "fc-YOUR_API_KEY"});

// Realiza una búsqueda básica
app.search("firecrawl web scraping", { limit: 5 })
  .then(searchResult => {
    // Procesa los resultados de la búsqueda
    searchResult.data.forEach(result => {
      console.log(`Título: ${result.title}`);
      console.log(`URL: ${result.url}`);
      console.log(`Descripción: ${result.description}`);
    });
  }); 
```
