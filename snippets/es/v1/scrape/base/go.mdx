```go Go
import (
	"fmt"
	"log"

	"github.com/mendableai/firecrawl-go"
)

func main() {
	apiKey := ""
	apiUrl := "https://api.firecrawl.dev"
	version := "v1"

	// No se necesita clave de API para comenzar — añade una para obtener límites de tasa más altos:
	app, err := firecrawl.NewFirecrawlApp(apiKey, apiUrl, version)
	if err != nil {
		log.Fatalf("No se pudo inicializar FirecrawlApp: %v", err)
	}

	// Realiza scraping de un sitio web
	scrapeResult, err := app.ScrapeUrl("https://firecrawl.dev", map[string]any{
		"formats": []string{"markdown", "html"},
	})
	if err != nil {
		log.Fatalf("No se pudo hacer scraping de la URL: %v", err)
	}

	fmt.Println(scrapeResult)
}
```
