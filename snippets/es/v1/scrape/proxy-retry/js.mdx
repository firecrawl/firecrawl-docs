```js Node
// npm install @mendable/firecrawl-js

import { FirecrawlApp } from '@mendable/firecrawl-js';

const app = new FirecrawlApp({ apiKey: 'YOUR_API_KEY' });

// Función para hacer scraping con lógica de reintentos
async function scrapeWithRetry(url) {
  try {
    // Primer intento con el proxy predeterminado
    const content = await app.scrapeUrl(url);
    
    // Verificar si se obtuvo un código de estado de error
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`Se obtuvo el código de estado ${statusCode}; reintentando con el proxy stealth`);
      // Reintentar con el proxy stealth
      return await app.scrapeUrl(url, {
        proxy: 'stealth'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`Error: ${error.message}`);
    // Reintentar con el proxy stealth en caso de excepción
    try {
      return await app.scrapeUrl(url, {
        proxy: 'stealth'
      });
    } catch (retryError) {
      console.error(`El proxy stealth también falló: ${retryError.message}`);
      throw retryError;
    }
  }
}

// Uso
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
