```bash cURL
# Primer intento con el proxy predeterminado
RESPONSE=$(curl -s -X POST https://api.firecrawl.dev/v1/scrape \
    -H 'Content-Type: application/json' \
    -H 'Authorization: Bearer YOUR_API_KEY' \
    -d '{
      "url": "https://example.com"
    }')

# Extraer el código de estado de la respuesta
STATUS_CODE=$(echo $RESPONSE | jq -r '.data.metadata.statusCode')

# Comprobar si el código de estado indica que debemos reintentar con stealth
if [[ "$STATUS_CODE" == "401" || "$STATUS_CODE" == "403" || "$STATUS_CODE" == "500" ]]; then
    echo "Se obtuvo el código de estado $STATUS_CODE; reintentando con el proxy stealth"
    
    # Reintentar con el proxy stealth
    curl -X POST https://api.firecrawl.dev/v1/scrape \
        -H 'Content-Type: application/json' \
        -H 'Authorization: Bearer YOUR_API_KEY' \
        -d '{
          "url": "https://example.com",
          "proxy": "stealth"
        }'
else
    # Mostrar la respuesta original
    echo $RESPONSE
fi
```
