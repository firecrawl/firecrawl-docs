```javascript JavaScript
import { Firecrawl } from 'firecrawl';

const app = new Firecrawl({
  // No se necesita API key para comenzar — agrega una para límites de tasa más altos:
  // apiKey: "fc-YOUR_API_KEY",
});

// Usa los datos en caché si tienen menos de 1 hora (3600000 ms)
// ¡Esto puede ser hasta significantly más rápido que un scrape nuevo!
const scrapeResult = await app.scrapeUrl('https://firecrawl.dev', {
  formats: ['markdown'],
  maxAge: 3600000 // 1 hora en milisegundos
});

console.log(scrapeResult.markdown);
```
