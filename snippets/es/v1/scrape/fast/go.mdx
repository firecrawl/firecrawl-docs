```go Go
package main

import (
    "fmt"
    "log"

    "github.com/mendableai/firecrawl-go"
)

func main() {
    // No se necesita API key para comenzar — agrega una para límites de tasa más altos:
    app, err := firecrawl.NewFirecrawlApp("")
    if err != nil {
        log.Fatalf("Error al inicializar FirecrawlApp: %v", err)
    }

    // Usa datos en caché si tienen menos de 1 hora (3600000 ms)
    // ¡Esto puede ser hasta un significantly más rápido que un scrape nuevo!
    scrapeParams := map[string]interface{}{
        "formats": []string{"markdown"},
        "maxAge":  3600000, // 1 hora en milisegundos
    }

    scrapeResult, err := app.ScrapeURL("https://firecrawl.dev", scrapeParams)
    if err != nil {
        log.Fatalf("Error al hacer scrape de la URL: %v", err)
    }

    fmt.Println(scrapeResult.Markdown)
}
```
