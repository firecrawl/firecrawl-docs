```rust Rust
use firecrawl::FirecrawlApp;
use std::collections::HashMap;

#[tokio::main]
async fn main() {
    // Inicializa FirecrawlApp con tu clave de API
    let app = FirecrawlApp::new("fc-YOUR_API_KEY").expect("No se pudo inicializar FirecrawlApp");

    // Usa datos en caché si tienen menos de 1 hora (3600000 ms)
    // ¡Esto puede ser hasta un 500% más rápido que un scrape desde cero!
    let mut scrape_params = HashMap::new();
    scrape_params.insert("formats", vec!["markdown"]);
    scrape_params.insert("maxAge", 3600000); // 1 hora en milisegundos

    let scrape_result = app
        .scrape_url("https://firecrawl.dev", Some(scrape_params))
        .await
        .expect("No se pudo hacer scraping de la URL");

    println!("{}", scrape_result.markdown.unwrap_or_default());
}
```
