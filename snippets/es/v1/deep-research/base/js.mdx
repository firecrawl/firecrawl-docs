```js Node
import FirecrawlApp from 'firecrawl';

// Inicializa el cliente
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// Define los parámetros de la investigación
const params = {
  maxDepth: 5,  // Número de iteraciones de investigación
  timeLimit: 180,  // Límite de tiempo en segundos
  maxUrls: 15  // Número máximo de URL a analizar
};

// Inicia la investigación con actualizaciones en tiempo real
const onActivity = (activity) => {
  console.log(`[${activity.type}] ${activity.message}`);
};

// Ejecuta una investigación profunda
const results = await firecrawl.deepResearch(
  '¿Cuáles son los últimos avances en computación cuántica?',
  params,
  onActivity
);

// Accede a los resultados de la investigación
console.log(`Análisis final: ${results.data.finalAnalysis}`);
console.log(`Fuentes: ${results.data.sources.length} referencias`); 
```
