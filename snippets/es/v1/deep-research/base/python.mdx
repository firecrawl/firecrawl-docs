```python Python
from firecrawl import FirecrawlApp

# Inicializa el cliente
firecrawl = FirecrawlApp(api_key="your_api_key")


# Inicia la investigación con actualizaciones en tiempo real
def on_activity(activity):
    print(f"[{activity['type']}] {activity['message']}")

# Ejecuta una investigación en profundidad
results = firecrawl.deep_research(
    query="¿Cuáles son los últimos avances en computación cuántica?",
    max_depth=5,
    time_limit=180,
    max_urls=15,
    on_activity=on_activity
)

# Accede a los resultados de la investigación.
print(f"Análisis final: {results['data']['finalAnalysis']}")

print(f"Fuentes: {len(results['data']['sources'])} referencias")


```
