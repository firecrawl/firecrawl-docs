```python Python
from firecrawl import FirecrawlApp

# Inicializa el cliente
firecrawl = FirecrawlApp(api_key="your_api_key")

# Consulta el estado de la investigación
status = firecrawl.check_deep_research_status("job_id")

# Print current progress
print(f"Estado: {status.status}")
print(f"Progreso: {status.current_depth}/{status.max_depth} iteraciones")

if status.status == 'completed':
print(f"Análisis final: {status.data.final_analysis}")
print(f"Fuentes encontradas: {len(status.data.sources)}")
```
