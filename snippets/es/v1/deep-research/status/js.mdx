```js Node
import FirecrawlApp from 'firecrawl';

// Inicializa el cliente
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// Comprueba el estado de la investigación
const status = await firecrawl.checkDeepResearchStatus('job_id');

// Muestra el progreso actual
console.log(`Estado: ${status.status}`);
console.log(`Progreso: ${status.currentDepth}/${status.maxDepth} iteraciones`);

if (status.status === 'completed') {
console.log(`Análisis final: ${status.data.finalAnalysis}`);
console.log(`Fuentes encontradas: ${status.data.sources.length}`);
} 
```
