```python Python
from firecrawl import Firecrawl

firecrawl = Firecrawl(api_key="fc-YOUR-API-KEY")

monitor = firecrawl.create_monitor(
    name="Blog monitor",
    schedule={"text": "every 30 minutes", "timezone": "UTC"},
    goal=(
        "Alert when a new blog post is published or the headline changes. "
        "Ignore unrelated navigation, footer, and formatting-only changes."
    ),
    targets=[
        {
            "type": "scrape",
            "urls": ["https://example.com/blog"],
        }
    ],
    notification={
        "email": {
            "enabled": True,
            "recipients": ["alerts@example.com"],
            "includeDiffs": True,
        }
    },
)

print(monitor.id)
```
