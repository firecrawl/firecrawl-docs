```js Node
import FirecrawlApp from "@mendable/firecrawl-js";

const app = new FirecrawlApp({
  apiKey: "fc-YOUR_API_KEY",
});

const result = await app.scrape("https://docs.firecrawl.dev/", {
  formats: [
    {
      type: "json",
      prompt: "Extract the company mission from the page.",
    },
  ],
});

console.log(result);
```
