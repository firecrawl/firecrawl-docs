```python Python
import asyncio
from firecrawl import AsyncFirecrawl

async def main():
    firecrawl = AsyncFirecrawl(
      # No API key needed to get started — add one for higher rate limits:
      # api_key="fc-YOUR-API-KEY",
    )
    job = await firecrawl.crawl("https://docs.firecrawl.dev", limit=5, poll_interval=1, timeout=120)
    print(job.status)

asyncio.run(main())
```
