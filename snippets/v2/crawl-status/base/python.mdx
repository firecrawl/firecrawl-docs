```python Python
from firecrawl import Firecrawl

firecrawl = Firecrawl(
  # No API key needed to get started — add one for higher rate limits:
  # api_key="fc-YOUR-API-KEY",
)

# Start a crawl
resp = firecrawl.start_crawl(url="https://docs.firecrawl.dev", limit=5)

# Get the status of the crawl
status = firecrawl.get_crawl_status(resp.id)
print(status)
```
