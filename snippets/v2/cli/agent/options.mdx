```bash CLI
# Use a specific model (spark-1-mini is default, spark-1-pro for higher accuracy)
firecrawl agent "Complex extraction task" --urls https://example.com --model spark-1-pro

# Set max credits to prevent runaway costs
firecrawl agent "Extract all products" --urls https://example.com --max-credits 100

# Check status of an existing job
firecrawl agent <job-id> --status

# Wait with custom polling interval and timeout
firecrawl agent "Extract data" --urls https://example.com --wait --poll-interval 10 --timeout 300

# Save output to file
firecrawl agent "Get pricing" --urls https://example.com --wait -o pricing.json --pretty
```
