```bash CLI
# Basic agent extraction
firecrawl agent "Find all pricing plans on this website" --urls https://example.com

# Extract structured data with a schema
firecrawl agent "Get company information" --urls https://example.com --schema '{"name": "string", "founded": "number"}'

# Use schema from a file
firecrawl agent "Extract product details" --urls https://example.com --schema-file schema.json

# Wait for results
firecrawl agent "Find contact information" --urls https://example.com --wait
```
