```js Node
const headers = { Authorization: "Bearer fc-YOUR_API_KEY" };

// 1. List the recorded tabs of the session
const meta = await fetch(
  `https://api.firecrawl.dev/v2/interact/${sessionId}/replay`,
  { headers },
).then(res => res.json());
const firstPage = meta.pages[0];

// 2. Fetch the HLS playlist for the first tab
const playlist = await fetch(
  `https://api.firecrawl.dev/v2/interact/${sessionId}/replay/${firstPage.pageId}`,
  { headers },
).then(res => res.text());

// The body is an HLS playlist. Each "https://..." line is a pre-signed
// segment URL valid for ~6 hours; hand the whole body to any HLS player.
console.log(playlist.split("\n")[0]); // "#EXTM3U"
```
