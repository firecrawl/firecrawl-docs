```bash cURL
# 1. List the recorded tabs of the session
curl "https://api.firecrawl.dev/v2/interact/$SESSION_ID/replay" \
  -H "Authorization: Bearer $FIRECRAWL_API_KEY"

# 2. Fetch the HLS playlist for a tab
curl "https://api.firecrawl.dev/v2/interact/$SESSION_ID/replay/0" \
  -H "Authorization: Bearer $FIRECRAWL_API_KEY"
```
