```js Node
import Firecrawl from '@mendable/firecrawl-js';

const app = new Firecrawl({ apiKey: 'fc-YOUR-API-KEY' });

// 1. Scrape the page
const result = await app.scrape('https://example.com', { formats: ['markdown'] });
const scrapeId = result.metadata?.scrapeId;
console.log(result.markdown);

// 2. Interact — ask the AI to do something on the page
const response = await app.interact(scrapeId, { prompt: 'What is the main heading on this page?' });
console.log(response.output); // "The main heading is 'Example Domain'."

// 3. Stop the session
await app.stopInteraction(scrapeId);
```
