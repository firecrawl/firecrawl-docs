```js Node
import Firecrawl from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: "fc-YOUR-API-KEY" });

// Search with language and country settings (German results from Germany)
const results = await firecrawl.search('web scraping tools', {
  limit: 5,
  lang: "de",
  country: "DE",
  location: "Germany"
});

// Process the results
console.log(results);
```
