```bash CLI
# Launch with custom TTL (10 minutes) and live view
firecrawl browser launch --ttl 600 --stream

# Launch with inactivity timeout
firecrawl browser launch --ttl 300 --ttl-inactivity 60

# Playwright Python — navigate, interact, extract
firecrawl browser execute '
await page.goto("https://news.ycombinator.com")
items = await page.query_selector_all(".titleline > a")
for item in items[:5]:
    print(await item.text_content())
'

# Playwright JavaScript — same page object
firecrawl browser execute --js '
await page.goto("https://example.com");
const title = await page.title();
console.log(title);
'

# Bash mode — runs locally with CDP_URL env injected
firecrawl browser execute --bash "agent-browser snapshot"

# Target a specific session
firecrawl browser execute --session <id> 'print(await page.title())'

# Save output to file
firecrawl browser execute 'print(await page.title())' -o result.txt

# Close a specific session
firecrawl browser close --session <id>

# JSON output for session list
firecrawl browser list --json --pretty
```
