```bash CLI
# Launch with custom TTL (10 minutes)
firecrawl browser launch --ttl 600

# Launch with inactivity timeout and live view
firecrawl browser launch --ttl 300 --ttl-inactivity 60 --stream

# Execute code in a specific session
firecrawl browser execute --session <id> 'print(await page.title())'

# Save execution output to file
firecrawl browser execute 'print(await page.title())' -o result.txt

# Close a specific session
firecrawl browser close --session <id>

# JSON output
firecrawl browser list --json --pretty
```
