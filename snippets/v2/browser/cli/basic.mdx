```bash CLI
# Launch a cloud browser session
firecrawl browser launch

# Execute Python code (default)
firecrawl browser execute 'await page.goto("https://example.com")
print(await page.title())'

# Execute JavaScript code
firecrawl browser execute --js 'await page.goto("https://example.com"); console.log(await page.title());'

# List active sessions
firecrawl browser list

# Close the active session
firecrawl browser close
```
