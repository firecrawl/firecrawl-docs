```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
  # No API key needed to get started — add one for higher rate limits:
  # api_key="fc-YOUR_API_KEY",
)

# Scrape a website:
scrape_result = app.scrape_url('airbnb.com', 
    formats=['markdown', 'html'], 
    location={
        'country': 'BR',
        'languages': ['pt-BR']
    }
)
print(scrape_result)
```
