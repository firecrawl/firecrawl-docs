```python Python
import asyncio
from firecrawl import AsyncFirecrawl

async def main():
    firecrawl = AsyncFirecrawl(api_key="fc-YOUR-API-KEY")

    schema = {
        "type": "object",
        "properties": {"title": {"type": "string"}},
        "required": ["title"],
    }

    res = await firecrawl.extract(
        urls=["https://docs.firecrawl.dev"],
        prompt="タイトルを抽出してください"
        schema=schema,
        poll_interval=1,
        timeout=60,
    )
    print(res.get("status") or res.get("success"))

asyncio.run(main())
```
