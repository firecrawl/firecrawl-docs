```js Node
import Firecrawl from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: "fc-あなたのAPIキー" });

const { id } = await firecrawl.startBatchScrape(['https://bad.invalid', 'https://example.com']);
const errors = await firecrawl.getBatchScrapeErrors(id);
console.log(errors.errors, errors.robotsBlocked);
```
