```js Node
// npm install @mendable/firecrawl-js

import { Firecrawl } from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: 'YOUR_API_KEY' });

// リトライロジック付きのスクレイプ関数
async function scrapeWithRetry(url) {
  try {
    // まずはデフォルトのプロキシで試す
    const content = await firecrawl.scrape(url);
    
    // エラーステータスコードか確認
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`ステータスコード ${statusCode} を検出。enhanced プロキシで再試行します`);
      // enhanced プロキシで再試行
      return await firecrawl.scrape(url, {
        proxy: 'enhanced'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`エラー: ${error.message}`);
    // 例外発生時はenhanced プロキシで再試行
    try {
      return await firecrawl.scrape(url, {
        proxy: 'enhanced'
      });
    } catch (retryError) {
      console.error(`enhanced プロキシでも失敗しました: ${retryError.message}`);
      throw retryError;
    }
  }
}

// 使用例
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
