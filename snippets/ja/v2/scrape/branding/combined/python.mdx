```python Python
from firecrawl import Firecrawl

firecrawl = Firecrawl(
  # 開始にAPIキーは不要です — レート制限を引き上げるには追加してください:
  # api_key='fc-YOUR_API_KEY',
)

    formats=['markdown', 'branding', 'screenshot']
)

print(result['markdown'])
print(result['branding'])
print(result['screenshot'])
```
