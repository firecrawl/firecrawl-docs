```python Python
from firecrawl import Firecrawl

firecrawl = Firecrawl(
  # 開始にAPIキーは不要です — レート制限を引き上げるには追加してください:
  # api_key="fc-YOUR-API-KEY",
)

doc = firecrawl.scrape('https://example.com',
    formats=['markdown'],
    location={
        'country': 'US',
        'languages': ['en']
    }
)

print(doc)
```
