```python Python
from firecrawl import Firecrawl

firecrawl = Firecrawl(api_key="fc-YOUR_API_KEY")

# ロケーション（ドイツ）を指定して検索
search_result = firecrawl.search(
    "web scraping tools",
    limit=5,
    location="Germany"
)

# 結果を処理
for result in search_result.data:
    print(f"タイトル: {result['title']}")
    print(f"URL: {result['url']}")
```
