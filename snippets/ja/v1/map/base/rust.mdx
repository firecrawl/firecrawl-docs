```rust Rust
use firecrawl::FirecrawlApp;

#[tokio::main]
async fn main() {
    // APIキーを使用して FirecrawlApp を初期化
    let app = FirecrawlApp::new("fc-YOUR_API_KEY").expect("FirecrawlApp の初期化に失敗しました");

    let map_result = app.map_url("https://firecrawl.dev", None).await;

    match map_result {
        Ok(data) => println!("マップされたURL: {:#?}", data),
        Err(e) => eprintln!("マッピングに失敗しました: {}", e),
    }
}
```
