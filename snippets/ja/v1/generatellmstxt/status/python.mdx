```python Python
from firecrawl import FirecrawlApp

# クライアントを初期化
firecrawl = FirecrawlApp(api_key="your_api_key")

# 非同期ジョブを作成
job = firecrawl.async_generate_llms_text(
    url="https://example.com",
)

if job.success:
    job_id = job.id

# LLMs.txt の生成状況を確認
status = firecrawl.check_generate_llms_text_status("job_id")

# 現在の状況を出力
print(f"Status: {status.status}")

if status.status == 'completed':
    print("LLMs.txt の内容:", status.data.llmstxt)
    if 'llmsfulltxt' in status.data:
        print("全文テキストの内容:", status.data.llmsfulltxt)
    print(f"処理済みのURL数: {len(status.data.processed_urls)}")
```
