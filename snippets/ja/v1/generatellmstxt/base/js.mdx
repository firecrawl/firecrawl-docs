```js Node
import FirecrawlApp from "firecrawl";

// クライアントを初期化する
const firecrawl = new FirecrawlApp({ apiKey: "your_api_key" });

// 生成用パラメータを定義
const params = {
  maxUrls: 2, // 分析するURLの最大数
  showFullText: true, // 結果に本文を含める
};

// ポーリングで LLMs.txt を生成する
const results = await firecrawl.generateLLMsText("https://example.com", params);

// 生成結果にアクセスする
if (results.success) {
  console.log(`ステータス: ${results.status}`);
  console.log(`生成されたデータ:`, results.data);
} else {
  console.error(`エラー: ${results.error || "不明なエラー"}`);
}
```
