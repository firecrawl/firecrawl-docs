```python Python
from firecrawl import FirecrawlApp

# クライアントを初期化する
firecrawl = FirecrawlApp(api_key="your_api_key")

# ポーリングで llms.txt を生成する
results = firecrawl.generate_llms_text(
    url="https://example.com",
    max_urls=2,
    show_full_text=True
)

# 生成結果にアクセスする
if results.success:
    print(f"ステータス: {results.status}")
    print(f"生成されたデータ: {results.data}")
else:
    print(f"エラー: {results.error}")
```
