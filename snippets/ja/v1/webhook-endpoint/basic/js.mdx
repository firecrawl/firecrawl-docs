const express = require('express');
const app = express();

app.post('/webhook', express.json(), (req, res) => {
  const { success, type, id, data, metadata, error } = req.body;
  
  switch (type) {
    case 'crawl.started':
    case 'batch_scrape.started':
      console.log(`${type.split('.')[0]} ${id} の開始`);
      break;
      
    case 'crawl.page':
    case 'batch_scrape.page':
      if (success && data.length > 0) {
        console.log(`ページをスクレイピングしました: ${data[0].metadata.sourceURL}`);
        // 取得したページデータを処理する
        processScrapedPage(data[0]);
      }
      break;
      
    case 'crawl.completed':
    case 'batch_scrape.completed':
      console.log(`${type.split('.')[0]} ${id} が正常に完了`);
      break;
      
    case 'crawl.failed':
    case 'batch_scrape.failed':
      console.error(`${type.split('.')[0]} ${id} が失敗しました: ${error}`);
      break;
  }
  
  // 受信の確認として常に 200 を返す
  res.status(200).send('OK');
});

function processScrapedPage(pageData) {
  // ここに処理ロジックを実装
  console.log('処理中:', pageData.metadata.title);
}

app.listen(3000, () => {
  console.log('Webhook サーバーがポート 3000 で待機中');
});
