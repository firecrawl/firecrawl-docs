```python Python
from firecrawl import FirecrawlApp

# APIキーを使って FirecrawlApp を初期化します

app = FirecrawlApp(api_key='your_api_key')

data = app.extract([
'https://nextbase.com/dash-cams/622gw-dash-cam'
], prompt="価格、機能、メリット・デメリット、レビューを含む、おすすめのドライブレコーダーに関する詳細を抽出してください。", enable_web_search=True)
print(data)
```
