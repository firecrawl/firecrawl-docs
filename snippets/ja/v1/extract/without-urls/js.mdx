```js Node
import { z } from "zod";

// コンテンツ抽出用のスキーマを定義
const schema = z.object({
  company_mission: z.string(),
});

const scrapeResult = await app.extract([], {
  prompt: "Firecrawl のウェブサイトから企業のミッションを抽出してください。",
  schema: schema
});

if (!scrapeResult.success) {
  throw new Error(`スクレイピングに失敗しました: ${scrapeResult.error}`)
}

console.log(scrapeResult.data);
```
