```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
    api_key="fc-YOUR_API_KEY"
)

# まず抽出ジョブを開始します
extract_job = app.async_extract([
    'https://docs.firecrawl.dev/*', 
    'https://firecrawl.dev/'
], prompt="これらのページから会社のミッションと機能を抽出してください。")

# 抽出ジョブのステータスを取得
job_status = app.get_extract_status(extract_job.id)

print(job_status)
# 出力例:
# id=None
# status='completed'
# expiresAt=datetime.datetime(...)
# success=True
# data=[{ ... }]
# error=None
# warning=None
# sources=None
```
