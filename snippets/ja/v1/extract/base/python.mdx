```python Python
from firecrawl import FirecrawlApp
from pydantic import BaseModel, Field

# APIキーを使って FirecrawlApp を初期化します
app = FirecrawlApp(api_key='your_api_key')

class ExtractSchema(BaseModel):
    company_mission: str
    supports_sso: bool
    is_open_source: bool
    is_in_yc: bool

data = app.extract([
  'https://docs.firecrawl.dev/*', 
  'https://firecrawl.dev/', 
  'https://www.ycombinator.com/companies/'
], prompt='ページから、企業のミッション、SSO対応の有無、オープンソースかどうか、Y Combinator に所属しているかを抽出してください。', schema=ExtractSchema.model_json_schema())
print(data)
```
