from pydantic import BaseModel

class ExtractSchema(BaseModel):
    company_mission: str

# 抽出対象のURLを指定する
urls = [
    'https://firecrawl.dev/*'
]

# 抽出用のプロンプトを指定する
prompt = 'ページから企業のミッションを抽出する。'

# 抽出を実行する
scrape_result = app.extract(urls, prompt=prompt, schema=ExtractSchema.model_json_schema())

# 抽出が成功したかを確認する
if not scrape_result.success:
    raise Exception(f"スクレイピングに失敗しました: {scrape_result.error}")

# 抽出結果のデータを出力する
print(scrape_result.data)
