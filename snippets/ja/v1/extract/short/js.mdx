```js Node
import { z } from "zod";

// コンテンツ抽出用のスキーマを定義
const schema = z.object({
  company_mission: z.string(),
});

const scrapeResult = await app.extract([
  "https://firecrawl.dev/*", 
], {
  prompt: "ページから企業のミッションを抽出せよ。",
  schema: schema
});

if (!scrapeResult.success) {
  throw new Error(`スクレイプに失敗しました: ${scrapeResult.error}`)
}

console.log(scrapeResult.data);
```
