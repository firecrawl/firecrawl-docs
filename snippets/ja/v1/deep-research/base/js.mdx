```js Node
import FirecrawlApp from 'firecrawl';

// クライアントを初期化する
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// リサーチ用のパラメータを定義
const params = {
  maxDepth: 5,  // リサーチの反復回数
  timeLimit: 180,  // 制限時間（秒）
  maxUrls: 15  // 解析するURLの最大数
};

// リアルタイム更新でリサーチを開始
const onActivity = (activity) => {
  console.log(`[${activity.type}] ${activity.message}`);
};

// 深堀りリサーチを実行
const results = await firecrawl.deepResearch(
  '量子コンピューティングの最新動向は？',
  params,
  onActivity
);

// リサーチ結果にアクセス
console.log(`最終的な分析: ${results.data.finalAnalysis}`);
console.log(`ソース: 参照 ${results.data.sources.length} 件`); 
```
