```python Python
from firecrawl import FirecrawlApp

# クライアントを初期化
firecrawl = FirecrawlApp(api_key="your_api_key")

# リサーチの状況を確認
status = firecrawl.check_deep_research_status("job_id")

# Print current progress
print(f"ステータス: {status.status}")
print(f"進捗: {status.current_depth}/{status.max_depth} 回の反復")

if status.status == 'completed':
    print(f"最終的な分析: {status.data.final_analysis}")
    print(f"発見したソース数: {len(status.data.sources)}") 
```
