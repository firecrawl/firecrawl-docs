```js Node.js
import FirecrawlApp from 'firecrawl';

// クライアントを初期化
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// リサーチのステータスを確認
const status = await firecrawl.checkDeepResearchStatus('job_id');

// 現在の進捗を表示
console.log(`ステータス: ${status.status}`);
console.log(`進捗: ${status.currentDepth}/${status.maxDepth} 回`);

if (status.status === 'completed') {
console.log(`最終結果: ${status.data.finalAnalysis}`);
console.log(`見つかったソース数: ${status.data.sources.length}`);
} 
```
