```bash cURL
# 開始するにAPIキーは不要です — より高いレート制限を利用するには -H "Authorization: Bearer YOUR_API_KEY" を追加してください:
curl -X POST https://api.firecrawl.dev/v1/scrape \
    -H 'Content-Type: application/json' \
    -d '{
        "url": "airbnb.com",
        "formats": ["markdown"],
        "location": {
            "country": "BR",
            "languages": ["pt-BR"]
        }
    }'
```
