```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# ウェブサイトをスクレイピングする:
scrape_result = app.scrape_url('firecrawl.dev',
  formats=['markdown', 'html'],
  agent={
    'model': 'FIRE-1',
    'prompt': '「次のページ」ボタンが無効になるまでクリックして商品一覧を移動し、各ページをスクレイピングしてください。'
  }
)

print(scrape_result)
```
