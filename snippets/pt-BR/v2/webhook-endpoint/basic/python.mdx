```python Python/Flask
from flask import Flask, request

app = Flask(__name__)

@app.post('/webhook/firecrawl')
def webhook():
    event = request.get_json(force=True, silent=True) or {}
    print('Webhook do Firecrawl:', event)
    return 'ok', 200

if __name__ == '__main__':
    app.run(port=3000)
```
