```python Python
# pip install firecrawl-py

from firecrawl import Firecrawl

firecrawl = Firecrawl(api_key="SUA_API_KEY")

# Primeiro, tente com o proxy básico
try:
    content = firecrawl.scrape("https://example.com")
    
    # Verifique se recebemos um código de status de erro
    status_code = content.get("metadata", {}).get("statusCode")
    if status_code in [401, 403, 500]:
        print(f"Recebemos o código de status {status_code}, tentando novamente com o proxy enhanced")
        # Tentar novamente com o proxy enhanced
        content = firecrawl.scrape("https://example.com", proxy="enhanced")
    
    print(content["markdown"])
except Exception as e:
    print(f"Erro: {e}")
    # Tentar novamente com o proxy enhanced em caso de exceção
    try:
        content = firecrawl.scrape("https://example.com", proxy="enhanced")
        print(content["markdown"])
    except Exception as e:
        print(f"O proxy enhanced também falhou: {e}")
```
