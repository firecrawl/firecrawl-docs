```js Node
// npm install @mendable/firecrawl-js

import { Firecrawl } from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: 'YOUR_API_KEY' });

// Função para fazer scraping com lógica de repetição (retry)
async function scrapeWithRetry(url) {
  try {
    // Primeiro tenta com o proxy padrão
    const content = await firecrawl.scrape(url);
    
    // Verifique se recebemos um código de status de erro
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`Recebemos o código de status ${statusCode}, tentando novamente com o proxy stealth`);
      // Tentar novamente com o proxy stealth
      return await firecrawl.scrape(url, {
        proxy: 'stealth'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`Erro: ${error.message}`);
    // Tentar novamente com o proxy stealth em caso de exceção
    try {
      return await firecrawl.scrape(url, {
        proxy: 'stealth'
      });
    } catch (retryError) {
      console.error(`O proxy stealth também falhou: ${retryError.message}`);
      throw retryError;
    }
  }
}

// Exemplo de uso
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
