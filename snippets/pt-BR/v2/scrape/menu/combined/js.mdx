```js Node
import { Firecrawl } from 'firecrawl';

const firecrawl = new Firecrawl({
  // Nenhuma API key necessária para começar — adicione uma para limite de taxa mais alto:
  // apiKey: "fc-YOUR-API-KEY",
});

const result = await firecrawl.scrape('https://example.com/restaurants/the-bistro/menu', {
    formats: ['markdown', 'menu']
});

console.log(result.markdown);
console.log(result.menu);
```
