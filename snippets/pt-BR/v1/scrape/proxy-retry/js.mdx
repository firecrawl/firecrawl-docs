```js Node
// npm install @mendable/firecrawl-js

import { FirecrawlApp } from '@mendable/firecrawl-js';

const app = new FirecrawlApp({ apiKey: 'YOUR_API_KEY' });

// Função para fazer scraping com lógica de repetição (retry)
async function scrapeWithRetry(url) {
  try {
    // Primeira tentativa com o proxy padrão
    const content = await app.scrapeUrl(url);
    
    // Verifique se recebemos um código de status de erro
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`Recebemos o código de status ${statusCode}; tentando novamente com proxy stealth`);
      // Tentar novamente com proxy stealth
      return await app.scrapeUrl(url, {
        proxy: 'stealth'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`Erro: ${error.message}`);
    // Tentar novamente com proxy stealth em caso de exceção
    try {
      return await app.scrapeUrl(url, {
        proxy: 'stealth'
      });
    } catch (retryError) {
      console.error(`O proxy stealth também falhou: ${retryError.message}`);
      throw retryError;
    }
  }
}

// Exemplo de uso
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
