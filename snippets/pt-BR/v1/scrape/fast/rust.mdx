```rust Rust
use firecrawl::FirecrawlApp;
use std::collections::HashMap;

#[tokio::main]
async fn main() {
    // Inicialize o FirecrawlApp com sua chave de API
    let app = FirecrawlApp::new("fc-YOUR_API_KEY").expect("Falha ao inicializar o FirecrawlApp");

    // Use os dados em cache se eles tiverem menos de 1 hora (3600000 ms)
    // Isso pode ser até 500% mais rápido do que uma coleta nova!
    let mut scrape_params = HashMap::new();
    scrape_params.insert("formats", vec!["markdown"]);
    scrape_params.insert("maxAge", 3600000); // 1 hora em milissegundos

    let scrape_result = app
        .scrape_url("https://firecrawl.dev", Some(scrape_params))
        .await
        .expect("Falha ao coletar a URL");

    println!("{}", scrape_result.markdown.unwrap_or_default());
}
```
