```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# Use o cache se ele tiver sido atualizado há menos de 1 hora (3600000 ms)
# Isso pode ser até 5x mais rápido do que uma raspagem do zero!
scrape_result = app.scrape_url(
    'https://firecrawl.dev', 
    params={
        'formats': ['markdown'],
        'maxAge': 3600000  # 1 hora em milissegundos
    }
)

print(scrape_result['markdown'])
```
