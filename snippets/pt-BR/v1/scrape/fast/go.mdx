package main

import (
    "fmt"
    "log"

    "github.com/mendableai/firecrawl-go"
)

func main() {
    app, err := firecrawl.NewFirecrawlApp("fc-YOUR_API_KEY")
    if err != nil {
        log.Fatalf("Falha ao inicializar o FirecrawlApp: %v", err)
    }

    // Use os dados em cache se tiverem menos de 1 hora (3600000 ms)
    // Isso pode ser 500% mais r√°pido do que uma raspagem do zero!
    scrapeParams := map[string]interface{}{
        "formats": []string{"markdown"},
        "maxAge":  3600000, // 1 hora em milissegundos
    }

    scrapeResult, err := app.ScrapeURL("https://firecrawl.dev", scrapeParams)
    if err != nil {
        log.Fatalf("Falha ao extrair a URL: %v", err)
    }

    fmt.Println(scrapeResult.Markdown)
}
