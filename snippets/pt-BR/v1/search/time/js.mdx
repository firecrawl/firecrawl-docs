```js Node.js
import FirecrawlApp from '@mendable/firecrawl-js';

// Inicialize o cliente com sua chave de API
const app = new FirecrawlApp({apiKey: "fc-YOUR_API_KEY"});

// Pesquise por resultados da última semana
app.search("técnicas mais recentes de web scraping", {
  limit: 5,
  tbs: "qdr:w"  // qdr:w = última semana
})
.then(searchResult => {
  // Processe os resultados
  searchResult.data.forEach(result => {
    console.log(`Título: ${result.title}`);
    console.log(`URL: ${result.url}`);
  });
}); 
```
