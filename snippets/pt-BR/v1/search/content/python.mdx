```python Python
from firecrawl import FirecrawlApp, ScrapeOptions

# Inicialize o cliente com sua chave de API
app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# Pesquise e extraia o conteúdo
search_result = app.search(
    "firecrawl web scraping",
    limit=3,
    scrape_options=ScrapeOptions(formats=["markdown", "links"])
)

# Processe os resultados
for result in search_result.data:
    print(f"Título: {result['title']}")
    print(f"URL: {result['url']}")
    print(f"Conteúdo: {result['markdown'][:150]}...")  # primeiros 150 caracteres
    print(f"Links: {', '.join(result['links'][:3])}...")  # primeiros 3 links
```
