```js Node.js
import FirecrawlApp from "firecrawl";

// Inicialize o cliente
const firecrawl = new FirecrawlApp({ apiKey: "your_api_key" });

// Crie uma tarefa assíncrona
const job = await firecrawl.asyncGenerateLLMsText("https://example.com", {
  maxUrls: 50,
  showFullText: true
});

// Verifique o status da geração do LLMs.txt
const status = await firecrawl.checkGenerateLLMsTextStatus(job.jobId);

// Exiba o status atual
console.log(`Status: ${status.status}`);

if (status.status === "completed") {
  console.log("Conteúdo do LLMs.txt:", status.data.llmstxt);
  if (status.data.llmsfulltxt) {
    console.log("Conteúdo do texto completo:", status.data.llmsfulltxt);
  }
  console.log(`URLs processadas: ${status.data.processedUrls.length}`);
}
```
