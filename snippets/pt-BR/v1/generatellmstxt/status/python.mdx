```python Python
from firecrawl import FirecrawlApp

# Inicialize o cliente
firecrawl = FirecrawlApp(api_key="your_api_key")

# Criar job assíncrono
job = firecrawl.async_generate_llms_text(
    url="https://example.com",
)

if job.success:
    job_id = job.id

# Verificar o status da geração do LLMs.txt
status = firecrawl.check_generate_llms_text_status("job_id")

# Exibir o status atual
print(f"Status: {status.status}")

if status.status == 'completed':
    print("Conteúdo do LLMs.txt:", status.data.llmstxt)
    if 'llmsfulltxt' in status.data:
        print("Conteúdo do texto completo:", status.data.llmsfulltxt)
    print(f"URLs processadas: {len(status.data.processed_urls)}")
```
