```python Python
from firecrawl import FirecrawlApp

# Inicialize o FirecrawlApp com sua chave de API
app = FirecrawlApp(api_key='your_api_key')

data = app.extract([
  'https://docs.firecrawl.dev/',
  'https://firecrawl.dev/'
], prompt="Extraia a missão da Firecrawl a partir da página.")
print(data)
```
