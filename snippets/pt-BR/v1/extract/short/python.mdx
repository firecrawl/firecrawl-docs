    company_mission: str

    'https://firecrawl.dev/*'
]

# Defina o prompt de extração
prompt = 'Extraia a missão da empresa da página.'

# Execute a extração
scrape_result = app.extract(urls, prompt=prompt, schema=ExtractSchema.model_json_schema())

# Verifique se a extração foi concluída com sucesso
if not scrape_result.success:

print(scrape_result.data)
