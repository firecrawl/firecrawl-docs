```python Python
from firecrawl import FirecrawlApp
from pydantic import BaseModel, Field

# Inicialize o FirecrawlApp com sua chave de API
app = FirecrawlApp(api_key='your_api_key')

class ExtractSchema(BaseModel):
    company_mission: str
    supports_sso: bool
    is_open_source: bool
    is_in_yc: bool

data = app.extract([
  'https://docs.firecrawl.dev/*', 
  'https://firecrawl.dev/', 
  'https://www.ycombinator.com/companies/'
], prompt='Extraia, a partir da página, a missão da empresa, se oferece suporte a SSO, se é de código aberto e se está no Y Combinator.', schema=ExtractSchema.model_json_schema())
print(data)
```
