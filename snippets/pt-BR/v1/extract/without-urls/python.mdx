from pydantic import BaseModel

class ExtractSchema(BaseModel):
    company_mission: str


# Defina o prompt para a extração
prompt = 'Extraia a missão da empresa no site da Firecrawl.'

# Execute a extração
scrape_result = app.extract(prompt=prompt, schema=ExtractSchema.model_json_schema())

# Verifique se a extração foi bem-sucedida
if not scrape_result.success:
    raise Exception(f"Falha ao extrair: {scrape_result.error}")

# Exiba os dados extraídos
print(scrape_result.data)
