```js Node
import FirecrawlApp from "@mendable/firecrawl-js";

const app = new FirecrawlApp({
apiKey: "fc-YOUR_API_KEY"
});

const scrapeResult = await app.extract([
'https://nextbase.com/dash-cams/622gw-dash-cam'
], {
prompt: "Extraia detalhes sobre as melhores câmeras veiculares, incluindo preços, recursos, prós e contras e avaliações.",
enableWebSearch: true // Ativa a pesquisa na web para melhor contexto
});

if (!scrapeResult.success) {
throw new Error(`Falha ao extrair dados: ${scrapeResult.error}`)
}

console.log(scrapeResult.data);
```
