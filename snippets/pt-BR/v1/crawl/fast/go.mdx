```go Go
package main

import (
    "fmt"
    "log"
    "github.com/firecrawl/firecrawl-go"
)

func main() {
    app, err := firecrawl.NewFirecrawlApp("fc-YOUR_API_KEY")
    if err != nil {
        log.Fatalf("Falha ao inicializar o FirecrawlApp: %v", err)
    }

    // Crawl com scraping em cache — 500% mais rápido para páginas acessadas recentemente
    crawlParams := map[string]interface{}{
        "limit": 100,
        "scrapeOptions": map[string]interface{}{
            "formats": []string{"markdown"},
            "maxAge":  3600000, // Usar dados em cache se tiver menos de 1 hora
        },
    }

    crawlResult, err := app.CrawlURL("https://firecrawl.dev", crawlParams)
    if err != nil {
        log.Fatalf("Falha ao fazer o crawl da URL: %v", err)
    }

    for _, page := range crawlResult.Data {
        fmt.Printf("URL: %s\n", page.Metadata["sourceURL"])
        if len(page.Markdown) > 200 {
            fmt.Printf("Conteúdo: %s...\n", page.Markdown[:200])
        } else {
            fmt.Printf("Conteúdo: %s\n", page.Markdown)
        }
    }
}
```
