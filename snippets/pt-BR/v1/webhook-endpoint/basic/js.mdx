const express = require('express');
const app = express();

app.post('/webhook', express.json(), (req, res) => {
  const { success, type, id, data, metadata, error } = req.body;
  
  switch (type) {
    case 'crawl.started':
    case 'batch_scrape.started':
      console.log(`${type.split('.')[0]} ${id} iniciado`);
      break;
      
    case 'crawl.page':
    case 'batch_scrape.page':
      if (success && data.length > 0) {
        console.log(`Página coletada: ${data[0].metadata.sourceURL}`);
        // Processar os dados da página coletada
        processScrapedPage(data[0]);
      }
      break;
      
    case 'crawl.completed':
    case 'batch_scrape.completed':
      console.log(`${type.split('.')[0]} ${id} concluído com sucesso`);
      break;
      
    case 'crawl.failed':
    case 'batch_scrape.failed':
      console.error(`${type.split('.')[0]} ${id} com falha: ${error}`);
      break;
  }
  
  // Sempre responda com 200 para confirmar o recebimento
  res.status(200).send('OK');
});
