# dentro de uma função assíncrona...
nest_asyncio.apply()

# Defina os manipuladores de eventos
def on_document(detail):
    print("DOC", detail)

def on_error(detail):
    print("ERR", detail['error'])

def on_done(detail):
    print("CONCLUÍDO", detail['status'])

    # Função para iniciar o processo de rastreamento e observação
async def start_crawl_and_watch():
    # Inicie o job de rastreamento e obtenha o observador
    watcher = app.crawl_url_and_watch('firecrawl.dev', limit=5)

    # Adicione ouvintes de eventos
    watcher.add_event_listener("document", on_document)
    watcher.add_event_listener("error", on_error)
    watcher.add_event_listener("done", on_done)

    # Inicie o observador
    await watcher.connect()

# Execute o loop de eventos
await start_crawl_and_watch()
