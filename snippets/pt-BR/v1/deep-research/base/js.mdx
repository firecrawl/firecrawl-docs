```js Node
import FirecrawlApp from 'firecrawl';

// Inicialize o cliente
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// Defina os parâmetros da pesquisa
const params = {
  maxDepth: 5,  // Número de iterações da pesquisa
  timeLimit: 180,  // Limite de tempo (em segundos)
  maxUrls: 15  // Número máximo de URLs a analisar
};

// Inicie a pesquisa com atualizações em tempo real
const onActivity = (activity) => {
  console.log(`[${activity.type}] ${activity.message}`);
};

// Execute uma pesquisa aprofundada
const results = await firecrawl.deepResearch(
  'Quais são os avanços mais recentes em computação quântica?',
  params,
  onActivity
);

// Acesse os resultados da pesquisa
console.log(`Análise final: ${results.data.finalAnalysis}`);
console.log(`Fontes: ${results.data.sources.length} referências`); 
```
