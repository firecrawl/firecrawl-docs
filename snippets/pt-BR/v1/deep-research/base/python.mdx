```python Python
from firecrawl import FirecrawlApp

# Inicialize o cliente
firecrawl = FirecrawlApp(api_key="your_api_key")


# Inicie a pesquisa com atualizações em tempo real
def on_activity(activity):
    print(f"[{activity['type']}] {activity['message']}")

# Execute uma pesquisa aprofundada
results = firecrawl.deep_research(
    query="Quais são os avanços mais recentes em computação quântica?",
    max_depth=5,
    time_limit=180,
    max_urls=15,
    on_activity=on_activity
)

# Acesse os resultados da pesquisa.
print(f"Análise final: {results['data']['finalAnalysis']}")

print(f"Fontes: {len(results['data']['sources'])} referências")


```
