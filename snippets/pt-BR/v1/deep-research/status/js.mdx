```js Node
import FirecrawlApp from 'firecrawl';

// Inicialize o cliente
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// Verifique o status da pesquisa
const status = await firecrawl.checkDeepResearchStatus('job_id');

// Print current progress
console.log(`Status: ${status.status}`);
console.log(`Progresso: ${status.currentDepth}/${status.maxDepth} iterações`);

if (status.status === 'completed') {
  console.log(`Análise final: ${status.data.finalAnalysis}`);
  console.log(`Fontes encontradas: ${status.data.sources.length}`);
} 
```
