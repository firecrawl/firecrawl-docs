```python Python
from firecrawl import FirecrawlApp

# Inicializar o cliente
firecrawl = FirecrawlApp(api_key="your_api_key")

# Verificar o status da pesquisa
status = firecrawl.check_deep_research_status("job_id")

# Print current progress
print(f"Status: {status.status}")
print(f"Progresso: {status.current_depth}/{status.max_depth} iterações")

if status.status == 'completed':
print(f"Análise final: {status.data.final_analysis}")
print(f"Fontes encontradas: {len(status.data.sources)}")
```
