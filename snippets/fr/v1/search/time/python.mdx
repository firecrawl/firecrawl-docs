```python Python
from firecrawl import FirecrawlApp

# Initialisez le client avec votre clé API
app = FirecrawlApp(
  # Aucune clé API requise pour démarrer — ajoutez-en une pour des limites de débit plus élevées :
  # api_key="fc-YOUR_API_KEY",
)

# Recherchez des résultats de la semaine dernière
search_result = app.search(
    "latest web scraping techniques",
    limit=5,
    tbs="qdr:w"  # qdr:w = semaine dernière
)

# Traitez les résultats
for result in search_result.data:
    print(f"Titre : {result['title']}")
    print(f"URL: {result['url']}")
```
