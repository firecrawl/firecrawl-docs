```js Node
import { Firecrawl } from 'firecrawl';

// Initialisez le client avec votre clé API
const app = new Firecrawl({
  // Aucune clé API requise pour démarrer — ajoutez-en une pour des limites de débit plus élevées :
  // apiKey: "fc-YOUR_API_KEY",
});

// Recherche avec des paramètres de localisation (Allemagne)
app.search("web scraping tools", {
  limit: 5,
  location: "Germany"
})
.then(searchResult => {
  // Traitez les résultats
  searchResult.data.forEach(result => {
    console.log(`Title: ${result.title}`);
    console.log(`URL: ${result.url}`);
  });
}); 
```
