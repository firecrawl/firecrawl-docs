```python Python
from firecrawl import FirecrawlApp, ScrapeOptions

# Initialisez le client avec votre clé API
app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# Rechercher et scraper du contenu
search_result = app.search(
    "firecrawl web scraping",
    limit=3,
    scrape_options=ScrapeOptions(formats=["markdown", "links"])
)

# Traiter les résultats
for result in search_result.data:
    print(f"Titre : {result['title']}")
    print(f"URL: {result['url']}")
    print(f"Contenu : {result['markdown'][:150]}...")  # premiers 150 caractères
    print(f"Liens : {', '.join(result['links'][:3])}...")  # 3 premiers liens
```
