```js Node
import FirecrawlApp from '@mendable/firecrawl-js';

// Initialisez le client avec votre clé d’API
const app = new FirecrawlApp({apiKey: "fc-YOUR_API_KEY"});

// Recherchez et scrappez du contenu
app.search("firecrawl web scraping", {
  limit: 3,
  scrapeOptions: {
    formats: ["markdown", "links"]
  }
})
.then(searchResult => {
  // Traitez les résultats
  searchResult.data.forEach(result => {
    console.log(`Titre : ${result.title}`);
    console.log(`URL: ${result.url}`);
    console.log(`Contenu : ${result.markdown?.substring(0, 150)}...`);
    console.log(`Liens : ${(result.links || []).slice(0, 3).join(', ')}...`);
  });
}); 
```
