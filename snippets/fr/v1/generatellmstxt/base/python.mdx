```python Python
from firecrawl import FirecrawlApp

# Initialiser le client
firecrawl = FirecrawlApp(api_key="your_api_key")

# Générer LLMs.txt avec sondage
results = firecrawl.generate_llms_text(
    url="https://example.com",
    max_urls=2,
    show_full_text=True
)

# Accéder aux résultats de la génération
if results.success:
    print(f"Statut : {results.status}")
    print(f"Données générées : {results.data}")
else:
    print(f"Erreur : {results.error}")
```
