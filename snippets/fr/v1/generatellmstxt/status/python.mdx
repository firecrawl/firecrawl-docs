```python Python
from firecrawl import FirecrawlApp

# Initialiser le client
firecrawl = FirecrawlApp(api_key="your_api_key")

# Créer une tâche asynchrone
job = firecrawl.async_generate_llms_text(
    url="https://example.com",
)

if job.success:
    job_id = job.id

# Vérifier l’état de génération de LLMs.txt
status = firecrawl.check_generate_llms_text_status("job_id")

# Afficher l'état en cours
print(f"Statut : {status.status}")

if status.status == 'completed':
    print("Contenu de LLMs.txt :", status.data.llmstxt)
    if 'llmsfulltxt' in status.data:
        print("Texte intégral :", status.data.llmsfulltxt)
    print(f"URL traitées : {len(status.data.processed_urls)}")
```
