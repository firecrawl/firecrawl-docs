```js Node.js
import FirecrawlApp from "firecrawl";

// Initialiser le client
const firecrawl = new FirecrawlApp({ apiKey: "your_api_key" });

// Créer une tâche asynchrone
const job = await firecrawl.asyncGenerateLLMsText("https://example.com", {
  maxUrls: 50,
  showFullText: true
});

// Vérifier l’état de génération de LLMs.txt
const status = await firecrawl.checkGenerateLLMsTextStatus(job.jobId);

// Afficher l’état actuel
console.log(`Statut : ${status.status}`);

if (status.status === "completed") {
  console.log("Contenu de LLMs.txt :", status.data.llmstxt);
  if (status.data.llmsfulltxt) {
    console.log("Contenu du texte intégral :", status.data.llmsfulltxt);
  }
  console.log(`URL traitées : ${status.data.processedUrls.length}`);
}
```
