```python Python
from firecrawl import FirecrawlApp
from pydantic import BaseModel, Field

# Initialisez FirecrawlApp avec votre clé d’API
app = FirecrawlApp(api_key='your_api_key')

class ExtractSchema(BaseModel):
    company_mission: str
    supports_sso: bool
    is_open_source: bool
    is_in_yc: bool

data = app.extract([
  'https://docs.firecrawl.dev/*', 
  'https://firecrawl.dev/', 
  'https://www.ycombinator.com/companies/'
], prompt='Extrait depuis la page la mission de l’entreprise, si elle prend en charge le SSO, si elle est open source et si elle figure dans Y Combinator.', schema=ExtractSchema.model_json_schema())
print(data)
```
