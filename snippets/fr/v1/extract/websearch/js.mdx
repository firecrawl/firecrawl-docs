```js Node
import FirecrawlApp from "@mendable/firecrawl-js";

const app = new FirecrawlApp({
apiKey: "fc-YOUR_API_KEY"
});

const scrapeResult = await app.extract([
'https://nextbase.com/dash-cams/622gw-dash-cam'
], {
prompt: "Extrait les détails sur les meilleures dashcams, y compris les prix, les fonctionnalités, les avantages et inconvénients, ainsi que les avis.",
enableWebSearch: true // Activer la recherche sur le web pour un meilleur contexte
});

if (!scrapeResult.success) {
throw new Error(`Échec de l’extraction : ${scrapeResult.error}`)
}

console.log(scrapeResult.data);
```
