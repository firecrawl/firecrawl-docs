```js Node
import { z } from "zod";

// Définir le schéma pour extraire le contenu
const schema = z.object({
  company_mission: z.string(),
});

const scrapeResult = await app.extract([
  "https://firecrawl.dev/*", 
], {
  prompt: "Extrait la mission de l’entreprise depuis la page.",
  schema: schema
});

if (!scrapeResult.success) {
  throw new Error(`Échec de l’extraction : ${scrapeResult.error}`)
}

console.log(scrapeResult.data);
```
