from pydantic import BaseModel

class ExtractSchema(BaseModel):
    company_mission: str

# Définir les URL dont extraire les données
urls = [
    'https://firecrawl.dev/*'
]

# Définir le prompt d’extraction
prompt = 'Extrait la mission de l’entreprise à partir de la page.'

# Lancer l’extraction
scrape_result = app.extract(urls, prompt=prompt, schema=ExtractSchema.model_json_schema())

# Vérifier si l’extraction a réussi
if not scrape_result.success:
    raise Exception(f"Échec de l’extraction : {scrape_result.error}")

# Afficher les données extraites
print(scrape_result.data)
