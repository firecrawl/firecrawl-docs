```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
    api_key="fc-YOUR_API_KEY"
)

# Démarrer d’abord un job d’extraction
extract_job = app.async_extract([
    'https://docs.firecrawl.dev/*', 
    'https://firecrawl.dev/'
], prompt="Extraire la mission de l’entreprise et ses fonctionnalités depuis ces pages.")

# Récupérer l’état du job d’extraction
job_status = app.get_extract_status(extract_job.id)

print(job_status)
# Exemple de sortie :
# id=None
# status='completed'
# expiresAt=datetime.datetime(...)
# success=True
# data=[{ ... }]
# error=None
# warning=None
# sources=None
```
