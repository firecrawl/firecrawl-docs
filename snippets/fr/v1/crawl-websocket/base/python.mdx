# à l’intérieur d’une fonction async...
nest_asyncio.apply()

# Définir les gestionnaires d’événements
def on_document(detail):
    print("DOC", detail)

def on_error(detail):
    print("ERR", detail['error'])

def on_done(detail):
    print("DONE", detail['status'])

    # Fonction pour démarrer le crawl et l’observateur
async def start_crawl_and_watch():
    # Lancer le job de crawl et récupérer l’observateur
    watcher = app.crawl_url_and_watch('firecrawl.dev', limit=5)

    # Ajouter des écouteurs d’événements
    watcher.add_event_listener("document", on_document)
    watcher.add_event_listener("error", on_error)
    watcher.add_event_listener("done", on_done)

    # Démarrer l’observateur
    await watcher.connect()

# Exécuter la boucle d’événements
await start_crawl_and_watch()
