```rust Rust
use firecrawl::FirecrawlApp;
use std::collections::HashMap;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let app = FirecrawlApp::new("fc-YOUR_API_KEY").expect("Échec de l’initialisation de FirecrawlApp");

    // Exploration avec mise en cache du scraping — 500 % plus rapide pour les pages récemment vues
    let mut scrape_options = HashMap::new();
    scrape_options.insert("formats", vec!["markdown"]);
    scrape_options.insert("maxAge", 3600000); // Utiliser les données mises en cache si elles ont moins d’une heure

    let mut crawl_params = HashMap::new();
    crawl_params.insert("limit", 100);
    crawl_params.insert("scrapeOptions", scrape_options);

    let crawl_result = app.crawl_url("https://firecrawl.dev", Some(crawl_params)).await?;

    for page in crawl_result.data {
        println!("URL: {}", page.metadata.get("sourceURL").unwrap_or(&"Inconnu".to_string()));
        let content = page.markdown.unwrap_or_default();
        if content.len() > 200 {
            println!("Contenu : {}...", &content[..200]);
        } else {
            println!("Contenu : {}", content);
        }
    }

    Ok(())
}
```
