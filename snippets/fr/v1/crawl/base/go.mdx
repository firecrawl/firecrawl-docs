```go Go
import (
	"fmt"
	"log"

	"github.com/firecrawl/firecrawl-go"
)

func main() {
	// Initialiser FirecrawlApp avec votre clé API
	apiKey := "fc-VOTRE_CLÉ_API"
	apiUrl := "https://api.firecrawl.dev"
	version := "v1"

	app, err := firecrawl.NewFirecrawlApp(apiKey, apiUrl, version)
	if err != nil {
		log.Fatalf("Impossible d’initialiser FirecrawlApp : %v", err)
	}

	// Crawler un site web
	crawlStatus, err := app.CrawlUrl("https://firecrawl.dev", map[string]any{
		"limit": 100,
		"scrapeOptions": map[string]any{
			"formats": []string{"markdown", "html"},
		},
	})
	if err != nil {
		log.Fatalf("Impossible d’envoyer la requête de crawl : %v", err)
	}

	fmt.Println(crawlStatus) 
}
```
