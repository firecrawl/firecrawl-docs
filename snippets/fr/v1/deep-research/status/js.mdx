```js Node
import FirecrawlApp from 'firecrawl';

// Initialiser le client
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// Vérifier l’état de la recherche
const status = await firecrawl.checkDeepResearchStatus('job_id');

// Print current progress
console.log(`Statut : ${status.status}`);
console.log(`Avancement : ${status.currentDepth}/${status.maxDepth} itérations`);

if (status.status === 'completed') {
console.log(`Analyse finale : ${status.data.finalAnalysis}`);
console.log(`Sources trouvées : ${status.data.sources.length}`);
} 
```
