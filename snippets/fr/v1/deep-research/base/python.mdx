```python Python
from firecrawl import FirecrawlApp

# Initialiser le client
firecrawl = FirecrawlApp(api_key="your_api_key")


# Démarrer la recherche avec des mises à jour en temps réel
def on_activity(activity):
    print(f"[{activity['type']}] {activity['message']}")

# Lancer une recherche approfondie
results = firecrawl.deep_research(
    query="Quelles sont les dernières avancées en calcul quantique ?",
    max_depth=5,
    time_limit=180,
    max_urls=15,
    on_activity=on_activity
)

# Accéder aux résultats de la recherche.
print(f"Analyse finale : {results['data']['finalAnalysis']}")

print(f"Sources : {len(results['data']['sources'])} références")


```
