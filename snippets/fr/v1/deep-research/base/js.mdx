```js Node.js
import FirecrawlApp from 'firecrawl';

// Initialiser le client
const firecrawl = new FirecrawlApp({ apiKey: 'your_api_key' });

// Définir les paramètres de recherche
const params = {
  maxDepth: 5,  // Nombre d’itérations de recherche
  timeLimit: 180,  // Limite de temps en secondes
  maxUrls: 15  // Nombre maximal d’URL à analyser
};

// Lancer la recherche avec des mises à jour en temps réel
const onActivity = (activity) => {
  console.log(`[${activity.type}] ${activity.message}`);
};

// Exécuter une recherche approfondie
const results = await firecrawl.deepResearch(
  'Quelles sont les dernières avancées en calcul quantique ?',
  params,
  onActivity
);

// Accéder aux résultats de la recherche
console.log(`Analyse finale : ${results.data.finalAnalysis}`);
console.log(`Sources : ${results.data.sources.length} références`); 
```
