```js Node.js
import FirecrawlApp, { ScrapeResponse } from '@mendable/firecrawl-js';

const app = new FirecrawlApp({apiKey: "fc-YOUR_API_KEY"});

// Récupérer le contenu d’un site web :
const scrapeResult = await app.scrapeUrl('firecrawl.dev', {
  formats: ['markdown', 'html'],
  agent: {
    model: 'FIRE-1',
    prompt: 'Parcourez les pages produits en cliquant sur le bouton « Page suivante » jusqu’à ce qu’il soit grisé/désactivé. Récupérez chaque page.'
  }  
}) as ScrapeResponse;

if (!scrapeResult.success) {
  throw new Error(`Échec de la récupération : ${scrapeResult.error}`)
}

console.log(scrapeResult)
```
