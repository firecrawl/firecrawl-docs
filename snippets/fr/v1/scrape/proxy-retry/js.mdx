```js Node
// npm install @mendable/firecrawl-js

import { FirecrawlApp } from '@mendable/firecrawl-js';

const app = new FirecrawlApp({ apiKey: 'YOUR_API_KEY' });

// Fonction de scraping avec logique de nouvelle tentative (retry)
async function scrapeWithRetry(url) {
  try {
    // Première tentative avec le proxy par défaut
    const content = await app.scrapeUrl(url);
    
    // Vérifier si un code d’erreur a été renvoyé
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`Code de statut ${statusCode} reçu, nouvelle tentative avec un proxy enhanced`);
      // Nouvelle tentative avec un proxy enhanced
      return await app.scrapeUrl(url, {
        proxy: 'enhanced'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`Error: ${error.message}`);
    // Nouvelle tentative avec un proxy enhanced en cas d’exception
    try {
      return await app.scrapeUrl(url, {
        proxy: 'enhanced'
      });
    } catch (retryError) {
      console.error(`Le proxy enhanced a aussi échoué : ${retryError.message}`);
      throw retryError;
    }
  }
}

// Exemple d’utilisation
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
