```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# Utilisez les données en cache si elles datent de moins d’1 heure (3 600 000 ms)
# Cela peut être 5× plus rapide qu’un nouveau crawl !
scrape_result = app.scrape_url(
    'https://firecrawl.dev', 
    params={
        'formats': ['markdown'],
        'maxAge': 3_600_000  # 1 heure en millisecondes
    }
)

print(scrape_result['markdown'])
```
