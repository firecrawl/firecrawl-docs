```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
  # Aucune clé API requise pour démarrer — ajoutez-en une pour des limites de débit plus élevées :
  # api_key="fc-YOUR_API_KEY",
)

# Utilise les données en cache si elles ont moins d'1 heure (3600000 ms)
# Cela peut être 500 % plus rapide qu'un scrape classique !
scrape_result = app.scrape_url(
    'https://firecrawl.dev', 
    params={
        'formats': ['markdown'],
        'max_age': 3600000  # 1 heure en millisecondes
    }
)

print(scrape_result['markdown'])
```
