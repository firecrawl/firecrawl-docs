```rust Rust
use firecrawl::FirecrawlApp;
use std::collections::HashMap;

#[tokio::main]
async fn main() {
    // Initialisez FirecrawlApp avec votre clé API
    let app = FirecrawlApp::new("fc-YOUR_API_KEY").expect("Échec de l’initialisation de FirecrawlApp");

    // Utilisez les données en cache si elles ont moins d’une heure (3600000 ms)
    // Cela peut être 5× plus rapide qu’un nouveau scrape !
    let mut scrape_params = HashMap::new();
    scrape_params.insert("formats", vec!["markdown"]);
    scrape_params.insert("maxAge", 3600000); // 1 heure en millisecondes

    let scrape_result = app
        .scrape_url("https://firecrawl.dev", Some(scrape_params))
        .await
        .expect("Échec du scraping de l’URL");

    println!("{}", scrape_result.markdown.unwrap_or_default());
}
```
