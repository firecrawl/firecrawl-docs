```js Node
// npm install firecrawl

import { Firecrawl } from 'firecrawl';

const app = new Firecrawl({
  // Aucune clé API requise pour démarrer — ajoutez-en une pour des limites de débit plus élevées :
  // apiKey: 'YOUR_API_KEY',
});

// Utilisation du proxy enhanced pour les sites dotés de solutions anti-bot avancées
const content = await app.scrapeUrl('https://example.com', {
  proxy: 'enhanced'
});

console.log(content.markdown);
```
