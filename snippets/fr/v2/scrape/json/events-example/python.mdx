```python Python
from firecrawl import Firecrawl
from pydantic import BaseModel

app = Firecrawl(api_key="fc-VOTRE-CLE-API")

class CompanyInfo(BaseModel):
    company_mission: str
    supports_sso: bool
    is_open_source: bool
    is_in_yc: bool

result = app.scrape(
    'https://firecrawl.dev/',
    formats=[{
        "type": "json",
        "schema": CompanyInfo.model_json_schema()
    }]
)

print(result)
```
