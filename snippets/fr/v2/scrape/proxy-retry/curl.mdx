```bash cURL
# Premier essai avec le proxy par défaut
RESPONSE=$(curl -s -X POST https://api.firecrawl.dev/v2/scrape \
    -H 'Content-Type: application/json' \
    -H 'Authorization: Bearer YOUR_API_KEY' \
    -d '{
      "url": "https://example.com"
    }')

# Extraire le code de statut de la réponse
STATUS_CODE=$(echo $RESPONSE | jq -r '.data.metadata.statusCode')

# Vérifier si le code de statut indique qu’il faut réessayer avec le proxy furtif
if [[ "$STATUS_CODE" == "401" || "$STATUS_CODE" == "403" || "$STATUS_CODE" == "500" ]]; then
    echo "Code de statut $STATUS_CODE reçu, nouvel essai avec le proxy furtif"
    
    # Réessayer avec le proxy furtif
    curl -X POST https://api.firecrawl.dev/v2/scrape \
        -H 'Content-Type: application/json' \
        -H 'Authorization: Bearer YOUR_API_KEY' \
        -d '{
          "url": "https://example.com",
          "proxy": "stealth"
        }'
else
    # Afficher la réponse d’origine
    echo $RESPONSE
fi
```
