```js Node
// npm install @mendable/firecrawl-js

import { Firecrawl } from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: 'VOTRE_CLE_API' });

// Fonction de scraping avec logique de reprise
async function scrapeWithRetry(url) {
  try {
    // Premier essai avec le proxy par défaut
    const content = await firecrawl.scrape(url);
    
    // Vérifiez si un code d’erreur a été renvoyé
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`Code de statut ${statusCode} reçu, nouvelle tentative avec le proxy stealth`);
      // Nouvelle tentative avec le proxy stealth
      return await firecrawl.scrape(url, {
        proxy: 'stealth'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`Erreur : ${error.message}`);
    // Nouvelle tentative avec le proxy stealth en cas d’exception
    try {
      return await firecrawl.scrape(url, {
        proxy: 'stealth'
      });
    } catch (retryError) {
      console.error(`Le proxy stealth a aussi échoué : ${retryError.message}`);
      throw retryError;
    }
  }
}

// Exemple d’utilisation
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
