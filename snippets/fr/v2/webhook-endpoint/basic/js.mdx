```js Node/Express
// Exemple de gestionnaire de webhook (Express)
import express from 'express';

const app = express();
app.use(express.json());

app.post('/webhook/firecrawl', (req, res) => {
  const event = req.body;
  console.log('Webhook Firecrawl :', event);
  res.status(200).send('ok');
});

app.listen(3000, () => console.log('À l’écoute sur 3000'));
```
