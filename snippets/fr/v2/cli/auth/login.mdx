```bash CLI
# Connexion interactive (ouvre le navigateur ou demande la clé API)
firecrawl login

# Connexion avec authentification par navigateur
firecrawl login --method browser

# Connexion directe avec clé API
firecrawl login --api-key fc-YOUR-API-KEY

# Ou définir via variable d'environnement
export FIRECRAWL_API_KEY=fc-YOUR-API-KEY
```
