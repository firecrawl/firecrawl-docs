```js Node.js
import Firecrawl from '@mendable/firecrawl-js';

const firecrawl = new Firecrawl({ apiKey: "fc-YOUR-API-KEY" });

// Recherche avec paramètres de localisation (Allemagne)
const results = await firecrawl.search('outils de web scraping', {
  limit: 5,
  location: "Allemagne"
});

// Traitement des résultats
console.log(results);
```
