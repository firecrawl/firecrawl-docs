## Overview
Use Go's standard library to make a Firecrawl scrape request over HTTP.

## Prerequisites
- A Firecrawl API key
- Go 1.20+

## Install
No extra packages required; this example uses the standard library.

## Configure your API key
Set your API key as an environment variable before running your program.

```bash
export FIRECRAWL_API_KEY="fc-your-api-key"
```

## Make your first request
```go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
    "os"
)

func main() {
    payload := map[string]any{
        "url": "https://example.com",
        "formats": []string{"markdown"},
    }

    body, _ := json.Marshal(payload)
    req, _ := http.NewRequest("POST", "https://api.firecrawl.dev/v2/scrape", bytes.NewReader(body))
    req.Header.Set("Authorization", "Bearer "+os.Getenv("FIRECRAWL_API_KEY"))
    req.Header.Set("Content-Type", "application/json")

    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    var result map[string]any
    _ = json.NewDecoder(resp.Body).Decode(&result)
    fmt.Println(result)
}
```

## Complete example
```go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "os"
)

func main() {
    payload := map[string]any{
        "url": "https://example.com",
        "formats": []string{"markdown", "html"},
    }

    body, _ := json.Marshal(payload)
    req, _ := http.NewRequest("POST", "https://api.firecrawl.dev/v2/scrape", bytes.NewReader(body))
    req.Header.Set("Authorization", "Bearer "+os.Getenv("FIRECRAWL_API_KEY"))
    req.Header.Set("Content-Type", "application/json")

    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    raw, _ := io.ReadAll(resp.Body)
    fmt.Println(string(raw))
}
```
